Quick Start

# Clone the repository
git clone https://github.com/ZoneCog/elizaos-cpp.git
cd elizaos-cpp

# Create build directory
mkdir build && cd build

# Configure and build
cmake ..
make -j$(nproc)

# Run tests
../cpp/tests/elizaos_tests
